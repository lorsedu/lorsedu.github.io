[tex]a_{n} =a_{1}+(n-1)d[/tex]

[tex]a_{n} =total\\a_{1}=first term\\d=common difference\\n=number of terms\\[/tex]

[tex]a_{35} =(-3)+(35-1)(-4)[/tex]
